using System;
namespace program
    {
        static void Main(string[] args)
        {
            int[] nums = { 5, 3, 8, 9, 2, 1 };
            int[] result = {0,0};
            for (int i = 0; i < nums.Length; i++)
            {
                if (i % 2 == 0) result[0] += nums[i];
                else result[1] += nums[i];
            }
            Console.WriteLine("{0} {1}", result[0] ,result[1]);
        }
}
